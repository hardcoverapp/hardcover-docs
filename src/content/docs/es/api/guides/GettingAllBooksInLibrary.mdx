## Get a list of books belonging to the current user

<GraphQLExplorer query={`
{
      list_books(
            where: {
                user_books: {
                    user_id: {_eq: ##USER_ID##}
                }
            },
            distinct_on: book_id
            limit: 5
            offset: 0
      ) {
        book {
              title
              pages
              release_date
        }
      }
}
`} description={`
    This query will return a list of books that the user has added to their collection.
`} presentation='json'/>
